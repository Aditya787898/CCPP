#include<stdio.h>

int main() 
{
   int x,y,z;
 
  printf("Enter the three angles\n");
 
  scanf("%d %d %d", &x , &y, &z);
 
  if(x==y&&y==z&&z==x)
  
 printf("The triangle is equilateral\n");
 
  else if(x!=y&&y!=z&&z!=x)
  
 printf("The triangle is scalene\n");
 
  else if(x==y||y==z)
  
 printf("The triangle is isosceles");

}
